  generateItem(itemsList:Array<IItem>):IItem {
    const item =  itemsList[Math.floor(Math.random()*itemsList.length)]

  visit() {
    if (this.visited === false) {
      this.visited = true
      this.content = this.getAppearance()
      this.onVisit.emit()
